FILE(GLOB res_subdirs RELATIVE ${CMAKE_CURRENT_LIST_DIR} ${CMAKE_CURRENT_LIST_DIR}/*)
FOREACH(res_subdir ${res_subdirs})
    IF(IS_DIRECTORY ${CMAKE_CURRENT_LIST_DIR}/${res_subdir})
        set(CURRENT_LIST_DIR_BACKUP_RES ${CMAKE_CURRENT_LIST_DIR})
        set(CMAKE_CURRENT_LIST_DIR ${CMAKE_CURRENT_LIST_DIR}/${res_subdir})
        include(${CMAKE_CURRENT_LIST_DIR}/CMakeLists.txt)
        set(CMAKE_CURRENT_LIST_DIR ${CURRENT_LIST_DIR_BACKUP_RES})
    ENDIF()
ENDFOREACH()
